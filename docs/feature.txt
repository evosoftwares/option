  final double latitude;

  final String? city;

      address: 'Rua Exemplo, 123',
      neighborhood: 'Centro',
      city: 'S√£o Bernardo do Campo',
      state: 'SP',
    );
  }

  String get fullAddress {
    final parts = [address];
    if (neighborhood != null) parts.add(neighborhood!);
